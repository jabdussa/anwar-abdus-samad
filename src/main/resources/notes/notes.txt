Rest API Queries

2. User Create, Get, Delete

   export HEADER="'Content-Type: application/json'"
   export USER_URL='http://localhost:9000/v0'
   export USER_DAT="'{"ip1": "8.8.8.8", "ip2": "8.8.4.4"}'"

   # CREATE
   curl -X POST -H $HEADER $USER_URL/users -d $USER_DAT

   # FIND ALL
   curl $USER_URL/users
